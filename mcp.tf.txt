resource "aws_apigatewayv2_api" "mcp_apigw" {
  name          = "mcp-ml-tools-api"
  protocol_type = "HTTP"
}

resource "aws_apigatewayv2_authorizer" "mcp_cognito_jwt" {
  api_id           = aws_apigatewayv2_api.mcp_apigw.id
  authorizer_type  = "JWT"
  identity_sources = ["$request.header.Authorization"]
  name             = "cognito-rbac"

  jwt_configuration {
    issuer   = "https://cognito-idp.us-west-2.amazonaws.com/${aws_cognito_user_pool.cognito_rbac_pool.id}"
    audience = [aws_cognito_user_pool_client.cognito_rbac_pool_client.id]
  }
}
